import java.util.*;

// Tax class
class Tax {
    private String name;
    private double rate;

    public Tax(String name, double rate) {
        this.name = name;
        this.rate = rate;
    }

    public double getRate() {
        return rate;
    }
}

// TaxGroup class
class TaxGroup {
    private String name;
    private List<Tax> taxes;

    public TaxGroup(String name, List<Tax> taxes) {
        this.name = name;
        this.taxes = taxes;
    }

    public double calculateTax(double price) {
        double total = 0;
        for (Tax t : taxes) {
            total += price * t.getRate() / 100.0;
        }
        return total;
    }
}

// Product class
class Product {
    private String name;
    private double price;

    public Product(String name, double price) {
        this.name = name;
        this.price = price;
    }

    public String getName() { return name; }
    public double getPrice() { return price; }
}

// Invoice class
class Invoice {
    private List<Product> products = new ArrayList<>();
    private TaxGroup taxGroup;

    public Invoice(TaxGroup taxGroup) {
        this.taxGroup = taxGroup;
    }

    public void addProduct(Product p) {
        products.add(p);
    }

    public void printInvoice() {
        System.out.printf("%-10s %-6s %-6s%n%n", "Product", "Price", "Tax");

        double totalPrice = 0;
        double totalTax = 0;

        for (Product p : products) {
            double tax = taxGroup.calculateTax(p.getPrice());
            totalPrice += p.getPrice();
            totalTax += tax;
            System.out.printf("%-10s %-6.0f %-6.1f%n", p.getName(), p.getPrice(), tax);
        }

        System.out.printf("%n%-10s %-6.0f %-6.1f%n", "Total", totalPrice, totalTax);
        System.out.printf("%nGrand Total : %.1f%n", totalPrice + totalTax);
    }
}

// Main class
public class InvoiceDemo {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        
        TaxGroup vatOnly = new TaxGroup("VAT Only", List.of(new Tax("VAT", 18)));

    
        Invoice invoice = new Invoice(vatOnly);

        System.out.print("How many products do you want to enter? ");
        int n = sc.nextInt();
        sc.nextLine(); 

        for (int i = 0; i < n; i++) {
            System.out.print("Enter product name: ");
            String name = sc.nextLine();
            System.out.print("Enter product price: ");
            double price = sc.nextDouble();
            sc.nextLine(); 

            invoice.addProduct(new Product(name, price));
        }

        System.out.println("\n----- INVOICE -----\n");
        invoice.printInvoice();
    }
}
